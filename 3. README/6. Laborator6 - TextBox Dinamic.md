```cpp
    private void tbPret_KeyPress(object sender, KeyPressEventArgs e)
        {
            if (e.KeyChar >= '0' && e.KeyChar <= '9' || e.KeyChar == (char)8 || e.KeyChar == ',')
                e.Handled = false;
            else
                e.Handled = true;
        }
```
